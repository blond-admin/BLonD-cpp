27 void Resonators::wake_calc(const f_vector_t &NewTimeArray)
28 {
29  /*
30  * Wake calculation method as a function of time.*
31  */
32  fTimeArray = NewTimeArray;
33  fWake.resize(fTimeArray.size());
34  std::fill_n(fWake.begin(), fWake.size(), 0);
35 
36  for (uint i = 0; i < fNResonators; ++i) {
37  double alpha = fOmegaR[i] / (2 * fQ[i]);
38  double omega_bar = std::sqrt(fOmegaR[i] * fOmegaR[i] - alpha * alpha);
39 
40  for (uint j = 0; j < fWake.size(); ++j) {
41  double temp = fTimeArray[j];
42  // util::dump(&temp, 1, "temp ");
43  int sign = (temp > 0) - (temp < 0);
44  fWake[j] += (sign + 1) * fRS[i] * alpha * std::exp(-alpha * temp) *
45  (std::cos(omega_bar * temp) -
46  alpha / omega_bar * std::sin(omega_bar * temp));
47  }
48  }
49 }

141 void TravelingWaveCavity::wake_calc(const f_vector_t &NewTimeArray)
142 {
143  /*
144  * Wake calculation method as a function of time.*
145  */
146 
147  fTimeArray = NewTimeArray;
148  fWake.resize(fTimeArray.size());
149  std::fill_n(fWake.begin(), fWake.size(), 0);
150 
151  for (uint i = 0; i < fNResonators; ++i) {
152  double a_tilde = faFactor[i] / (2 * constant::pi);
153  for (uint j = 0; j < fWake.size(); ++j) {
154  if (fTimeArray[j] <= a_tilde) {
155  fWake[j] += (mymath::sign(fTimeArray[j]) + 1) *
156  2. * fRS[i] / a_tilde *
157  (1. - fTimeArray[j] / a_tilde) *
158  std::cos(2. * constant::pi * fFrequencyR[i]
159  * fTimeArray[j]);
160  }
161  }
162  }

193  );
