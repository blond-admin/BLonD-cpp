23 void Resonators::wake_calc(const f_vector_t& NewTimeArray) {
24  /*
25  * Wake calculation method as a function of time.*
26  */
27  fTimeArray = NewTimeArray;
28  fWake.resize(fTimeArray.size());
29  std::fill_n(fWake.begin(), fWake.size(), 0);
30 
31  for (uint i = 0; i < fNResonators; ++i) {
32  double alpha = fOmegaR[i] / (2 * fQ[i]);
33  double omega_bar = std::sqrt(fOmegaR[i] * fOmegaR[i] - alpha * alpha);
34 
35  for (uint j = 0; j < fWake.size(); ++j) {
36  double temp = fTimeArray[j];
37  // util::dump(&temp, 1, "temp ");
38  int sign = (temp > 0) - (temp < 0);
39  fWake[j] += (sign + 1) * fRS[i] * alpha * std::exp(-alpha * temp) *
40  (std::cos(omega_bar * temp) -
41  alpha / omega_bar * std::sin(omega_bar * temp));
42  }
43  }
44 }

46 void Resonators::imped_calc(const f_vector_t& NewFrequencyArray) {
47  /*
48  * Impedance calculation method as a function of frequency.*
49  */
50 
51  fFreqArray = NewFrequencyArray;
52  fImpedance.resize(fFreqArray.size());
53  std::fill_n(fImpedance.begin(), fImpedance.size(), complex_t(0, 0));
54 
55  for (uint i = 0; i < fNResonators; ++i) {
56  for (uint j = 1; j < fImpedance.size(); ++j) {
57  fImpedance[j] +=
58  complex_t(fRS[i], 0) /

60  fFrequencyR[i] / fFreqArray[j]));
