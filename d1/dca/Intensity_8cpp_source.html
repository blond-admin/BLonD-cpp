24 void Resonators::wake_calc(const f_vector_t& NewTimeArray) {
25  /*
26  * Wake calculation method as a function of time.*
27  */
28  fTimeArray = NewTimeArray;
29  fWake.resize(fTimeArray.size());
30  std::fill_n(fWake.begin(), fWake.size(), 0);
31  // util::dump(&fWake[0], 10, "start wake ");
32 
33  for (uint i = 0; i < fNResonators; ++i) {
34  ftype alpha = fOmegaR[i] / (2 * fQ[i]);
35  ftype omega_bar = std::sqrt(fOmegaR[i] * fOmegaR[i] - alpha * alpha);
36 
37  for (uint j = 0; j < fWake.size(); ++j) {
38  ftype temp = fTimeArray[j];
39  // util::dump(&temp, 1, "temp ");
40  int sign = (temp > 0) - (temp < 0);
41  fWake[j] += (sign + 1) * fRS[i] * alpha * std::exp(-alpha * temp) *
42  (std::cos(omega_bar * temp) -
43  alpha / omega_bar * std::sin(omega_bar * temp));
44  }
45  }
46 }
