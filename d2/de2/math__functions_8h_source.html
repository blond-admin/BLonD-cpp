110  const int N = x.size();
111  if (N == 0) return;

113  // const auto max = xp.back();

115 
116  // #pragma omp parallel for
117  for (int i = 0; i < N; ++i) {
118  int pos = std::lower_bound(xp.begin(), xp.end(), x[i]) - xp.begin();
119  if (pos == M)
120  y[i] = right;
121  else if (xp[pos] == x[i])
122  y[i] = yp[pos];
123  else if (pos == 0)
124  y[i] = left;
125  else {
126  y[i] = yp[pos - 1] +
