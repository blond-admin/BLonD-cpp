151 void LHCNoiseFB::fwhm_multi_bunch() {
152 
153  // Multi-bunch FWHM bunch length calculation with interpolation.*
154 
155  auto Slice = Context::Slice;
156  auto RfP = Context::RfP;
157 
158  // Find correct RF buckets
159  auto phi_RF = RfP->phi_RF[0][RfP->counter];
160  auto omega_RF = RfP->omega_RF[0][RfP->counter];
161  // std::cout << phi_RF << "\n";
162  // std::cout << omega_RF << "\n";
163  f_vector_t bucket_min(fBunchPattern.size());
164  for (uint i = 0; i < bucket_min.size(); ++i)
165  bucket_min[i] =
166  (phi_RF + 2 * constant::pi * fBunchPattern[i]) / omega_RF;
167 
168  f_vector_t bucket_max(bucket_min.size());
169  for (uint i = 0; i < bucket_min.size(); ++i)
170  bucket_max[i] = bucket_min[i] + 2 * constant::pi / omega_RF;
171 
172  // Bunch-by-bunch FWHM bunch length
173  for (uint i = 0; i < fBunchPattern.size(); ++i) {
174  uint_vector_t bind;
175  for (int j = 0; j < Slice->n_slices; ++j) {
176  if ((Slice->bin_centers[j] - bucket_min[i]) *
177  (Slice->bin_centers[j] - bucket_max[i]) < 0)
178  bind.push_back(j);
179  }
180 
181  ftype hheight = 0;
182  for (const auto& j : bind) {
183  if (Slice->n_macroparticles[j] > hheight)
184  hheight = Slice->n_macroparticles[j];
185  }
186 
187  hheight = hheight / 2;
188 
189  uint_vector_t index;
190  auto k = 0;
191  for (const auto &j : bind) {
192  if (Slice->n_macroparticles[j] > hheight)
193  index.push_back(bind[k]);
194  k++;
195  }
196  // std::cout << "index size = " << index.size() << "\n";
197  if (index.empty()) {
198  // std::cerr << "[LHCNoiseFB] ERROR! index vector should have at least one element\n";
199  continue;
200  }
201  fBlMeasBBB[i] = fwhm_interpolation(index, hheight);
202  }
203 
204  fBlMeas = mymath::mean(fBlMeasBBB.data(), fBlMeasBBB.size());
205 }
