10 void linear_interp_kick(
11  const double *__restrict beam_dt,
12  double *__restrict beam_dE,
13  const double *__restrict voltage_array,
14  const double *__restrict bin_centers,
15  const int n_slices,
16  const int n_macroparticles)
17 {
18 
19  const double binFirst = bin_centers[0];
20  const double binLast = bin_centers[n_slices - 1];
21  const double inv_bin_width = (n_slices - 1) / (binLast - binFirst);
22 
23  #pragma omp parallel for
24  for (int i = 0; i < n_macroparticles; ++i) {
25  const double a = beam_dt[i];
26  const int ffbin = static_cast<int>((a - binFirst) * inv_bin_width);
27  const double voltageKick =
28  ((a < binFirst) || (a > binLast))
29  ? 0.0
30  : voltage_array[ffbin] +
31  (a - bin_centers[ffbin]) *
32  (voltage_array[ffbin + 1] - voltage_array[ffbin]) *
33  inv_bin_width;
34  beam_dE[i] += voltageKick;
35  }
36 }

114 
115  if (fTimeOrFreq == freq_domain) {

324 f_vector_t InducedVoltageFreq::induced_voltage_generation(Beams *beam,
325  uint length)
326 {
327  // Method to calculate the induced voltage from the inverse FFT of the
328  // impedance times the spectrum (fourier convolution).
329 
330  if (fRecalculationImpedance)
331  sum_impedances(fFreqArray);
332 
333  fSlices->beam_spectrum_generation(fNFFTSampling);
334  const auto n = fImpedanceSourceList.size();
335  const auto factor = -beam->charge * constant::e * beam->ratio *
336  fSlices->fBeamSpectrumFreq[1] * 2 *
337  (fSlices->fBeamSpectrum.size() - 1);
338 
339  if (fSaveIndividualVoltages) {
340 
341  for (uint i = 0; i < n; ++i) {
342  f_vector_t res;
343  complex_vector_t in(fSlices->fBeamSpectrum.size());
344 
345  for (uint j = 0; j < in.size(); ++j) {
346  in[j] = fMatrixSaveIndividualImpedances[j * n + i] *
347  fSlices->fBeamSpectrum[j];
348  }
349 
350  fft::irfft(in, res, 0, Context::n_threads);
351 
352  assert((int)res.size() >= fSlices->n_slices);
353 
354  res.resize(fSlices->n_slices);
355  res *= factor;
356 
357  for (int j = 0; j < fSlices->n_slices; ++j) {
358  fMatrixSaveIndividualVoltages[j * n + i] = res[j];
359  }
360  }
361 
362  fInducedVoltage.clear();
363  fInducedVoltage.resize(fSlices->fBeamSpectrum.size());
364  for (uint i = 0; i < fSlices->fBeamSpectrum.size(); ++i) {
365  double sum = 0.0;
366  for (uint j = 0; j < n; ++j) {
367  sum += fMatrixSaveIndividualVoltages[i * n + j];
368  }
369  fInducedVoltage[i] = sum;
370  }
371 
372  return f_vector_t();
373 
374  } else {
375  f_vector_t res;
376  complex_vector_t in(fSlices->fBeamSpectrum.size());
377  for (uint j = 0; j < in.size(); ++j) {
378  in[j] = fTotalImpedance[j] * fSlices->fBeamSpectrum[j];
379  }
380 
381  fft::irfft(in, res, 0, Context::n_threads);
382  assert((int)res.size() >= fSlices->n_slices);
383 
384  res.resize(fSlices->n_slices);
385  res *= factor;
386 
387  fInducedVoltage = res;
388 
389  if (length > 0) {
390  if (length > res.size())
391  res.resize(length, 0);
392  else
393  res.resize(length);
394  }
395 
396  return res;
397  }
