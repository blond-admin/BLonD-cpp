13 void blond::linear_interp_kick(
14  const double *__restrict beam_dt,
15  double *__restrict beam_dE,
16  const double *__restrict voltage_array,
17  const double *__restrict bin_centers,
18  const int n_slices,
19  const int n_macroparticles)
20 {
21 
22  const double binFirst = bin_centers[0];
23  const double binLast = bin_centers[n_slices - 1];
24  const double inv_bin_width = (n_slices - 1) / (binLast - binFirst);
25 
26  #pragma omp parallel for
27  for (int i = 0; i < n_macroparticles; ++i) {
28  const double a = beam_dt[i];
29  const int ffbin = static_cast<int>((a - binFirst) * inv_bin_width);
30  const double voltageKick =
31  ((a < binFirst) || (a > binLast))
32  ? 0.0
33  : voltage_array[ffbin] +
34  (a - bin_centers[ffbin]) *
35  (voltage_array[ffbin + 1] - voltage_array[ffbin]) *
36  inv_bin_width;
37  beam_dE[i] += voltageKick;
38  }
39 }

109  int length)
110 {
111 
112  // Method to calculate the induced voltage from wakes with convolution.*
113  f_vector_t inducedVoltage;
114 
115  const double factor = -beam->charge * constant::e * beam->intensity
116  / beam->n_macroparticles;
117 
118  if (fTimeOrFreq == freq_domain) {
119  auto in1 = fSlices->n_macroparticles;
120  auto in2 = fTotalWake;
121 
122  fft::convolution_with_ffts(in1, in2, inducedVoltage);
123  inducedVoltage *= factor;

126 
127  inducedVoltage.resize(fTotalWake.size() + fSlices->n_slices - 1);
128 
129  mymath::convolution(fTotalWake.data(), fTotalWake.size(),
130  fSlices->n_macroparticles.data(),
131  fSlices->n_macroparticles.size(),
132  inducedVoltage.data());
133  inducedVoltage *= factor;
134 
135  } else {

137  exit(-1);

327 f_vector_t InducedVoltageFreq::induced_voltage_generation(Beams *beam,
328  int length)
329 {
330  // Method to calculate the induced voltage from the inverse FFT of the
331  // impedance times the spectrum (fourier convolution).
332 
333  if (fRecalculationImpedance)
334  sum_impedances(fFreqArray);
335 
336  fSlices->beam_spectrum_generation(fNFFTSampling);
337  const int n = fImpedanceSourceList.size();
338  const double factor = -beam->charge * constant::e * beam->ratio *
339  fSlices->fBeamSpectrumFreq[1] * 2 *
340  (fSlices->fBeamSpectrum.size() - 1);
341 
342  if (fSaveIndividualVoltages) {
343 
344  for (int i = 0; i < n; ++i) {
345  f_vector_t res;
346  complex_vector_t in(fSlices->fBeamSpectrum.size());
347 
348  for (uint j = 0; j < in.size(); ++j) {
349  in[j] = fMatrixSaveIndividualImpedances[j * n + i] *
350  fSlices->fBeamSpectrum[j];
351  }
352 
353  fft::irfft(in, res, 0, Context::n_threads);
354 
355  assert((int)res.size() >= fSlices->n_slices);
356 
357  res.resize(fSlices->n_slices);
358  res *= factor;
359 
360  for (int j = 0; j < fSlices->n_slices; ++j) {
361  fMatrixSaveIndividualVoltages[j * n + i] = res[j];
362  }
363  }
364 
365  fInducedVoltage.clear();
366  fInducedVoltage.resize(fSlices->fBeamSpectrum.size());
367  for (uint i = 0; i < fSlices->fBeamSpectrum.size(); ++i) {
368  double sum = 0.0;
369  for (int j = 0; j < n; ++j) {
370  sum += fMatrixSaveIndividualVoltages[i * n + j];
371  }
372  fInducedVoltage[i] = sum;
373  }
374 
375  return f_vector_t();
376 
377  } else {
378  f_vector_t res;
379  complex_vector_t in(fSlices->fBeamSpectrum.size());
380  for (uint j = 0; j < in.size(); ++j) {
381  in[j] = fTotalImpedance[j] * fSlices->fBeamSpectrum[j];
382  }
383 
384  fft::irfft(in, res, 0, Context::n_threads);
385  assert((int)res.size() >= fSlices->n_slices);
386 
387  res.resize(fSlices->n_slices);
388  res *= factor;
389 
390  fInducedVoltage = res;
391 
392  if (length > 0) {
393  if (length > (int)res.size())
394  res.resize(length, 0);
395  else
396  res.resize(length);
397  }
398 
399  return res;
400  }
