11  const double *__restrict beam_dt,
12  double *__restrict beam_dE,
13  const double *__restrict voltage_array,
14  const double *__restrict bin_centers,
15  const int n_slices,
16  const int n_macroparticles)
17 {
18 
19  const double binFirst = bin_centers[0];
20  const double binLast = bin_centers[n_slices - 1];
21  const double inv_bin_width = (n_slices - 1) / (binLast - binFirst);
22 
23  #pragma omp parallel for
24  for (int i = 0; i < n_macroparticles; ++i) {
25  const double a = beam_dt[i];
26  const int ffbin = static_cast<int>((a - binFirst) * inv_bin_width);
27  const double voltageKick =
28  ((a < binFirst) || (a > binLast))
29  ? 0.0
30  : voltage_array[ffbin] +
31  (a - bin_centers[ffbin]) *
32  (voltage_array[ffbin + 1] - voltage_array[ffbin]) *
33  inv_bin_width;
34  beam_dE[i] += voltageKick;
35  }
36 }

114 f_vector_t InducedVoltageTime::induced_voltage_generation(Beams *beam,
115  uint length)
116 {
117 
118  // Method to calculate the induced voltage from wakes with convolution.*
119  f_vector_t inducedVoltage;
120 
121  const double factor = -beam->charge * constant::e * beam->intensity
122  / beam->n_macroparticles;
123 
124  if (fTimeOrFreq == freq_domain) {
125  f_vector_t in1(fSlices->n_slices);
126  std::copy(fSlices->n_macroparticles.begin(),
127  fSlices->n_macroparticles.end(), in1.begin());
128 
129  f_vector_t in2 = fTotalWake;
130 
131  fft::convolution_with_ffts(in1, in2, inducedVoltage);
132 
133  std::transform(inducedVoltage.begin(), inducedVoltage.end(),
134  inducedVoltage.begin(),
135  std::bind1st(std::multiplies<double>(), factor));
136 
137  } else if (fTimeOrFreq == time_domain) {
138  f_vector_t temp(fSlices->n_slices);
139  std::copy(fSlices->n_macroparticles.begin(),
140  fSlices->n_macroparticles.end(), temp.begin());
141 
142  inducedVoltage.resize(fTotalWake.size() + temp.size() - 1);
143 
144  mymath::convolution(fTotalWake.data(), fTotalWake.size(), temp.data(),
145  temp.size(), inducedVoltage.data());
146 
147  std::transform(inducedVoltage.begin(), inducedVoltage.end(),
148  inducedVoltage.begin(),
149  std::bind1st(std::multiplies<double>(), factor));
150 
151  } else {
152  dprintf("Error: Only freq_domain or time_domain are allowed\n");
153  exit(-1);
154  }
155 
156  fInducedVoltage = inducedVoltage;
157  fInducedVoltage.resize((uint)fSlices->n_slices);
158 
159  if (length > 0)
160  inducedVoltage.resize(length, 0);
161 
162  return inducedVoltage;
163 }

351 f_vector_t InducedVoltageFreq::induced_voltage_generation(Beams *beam,
352  uint length)
353 {
354  // Method to calculate the induced voltage from the inverse FFT of the
355  // impedance times the spectrum (fourier convolution).
356 
357  if (fRecalculationImpedance)
358  sum_impedances(fFreqArray);
359 
360  fSlices->beam_spectrum_generation(fNFFTSampling);
361  // std::cout << "fNFFTSampling : " << fNFFTSampling << "\n";
362  const auto n = fImpedanceSourceList.size();
363  const auto factor = -beam->charge * constant::e * beam->ratio *
364  fSlices->fBeamSpectrumFreq[1] * 2 *
365  (fSlices->fBeamSpectrum.size() - 1);
366 
367  if (fSaveIndividualVoltages) {
368 
369  for (uint i = 0; i < n; ++i) {
370  f_vector_t res;
371  complex_vector_t in(fSlices->fBeamSpectrum.size());
372 
373  for (uint j = 0; j < in.size(); ++j) {
374  in[j] = fMatrixSaveIndividualImpedances[j * n + i] *
375  fSlices->fBeamSpectrum[j];
376  }
377 
378  fft::irfft(in, res, 0, Context::n_threads);
379 
380  assert((int)res.size() >= fSlices->n_slices);
381 
382  res.resize(fSlices->n_slices);
383 
384  std::transform(res.begin(), res.end(), res.begin(),
385  std::bind1st(std::multiplies<double>(), factor));
386 
387  for (int j = 0; j < fSlices->n_slices; ++j) {
388  fMatrixSaveIndividualVoltages[j * n + i] = res[j];
389  }
390  }
391 
392  fInducedVoltage.clear();
393  fInducedVoltage.resize(fSlices->fBeamSpectrum.size());
394  for (uint i = 0; i < fSlices->fBeamSpectrum.size(); ++i) {
395  double sum = 0.0;
396  for (uint j = 0; j < n; ++j) {
397  sum += fMatrixSaveIndividualVoltages[i * n + j];
398  }
399  fInducedVoltage[i] = sum;
400  }
401 
402  return f_vector_t();
403 
404  } else {
405  f_vector_t res;
406  complex_vector_t in(fSlices->fBeamSpectrum.size());
407  for (uint j = 0; j < in.size(); ++j) {
408  in[j] = fTotalImpedance[j] * fSlices->fBeamSpectrum[j];
409  }
410 
411  fft::irfft(in, res, 0, Context::n_threads);
412  // std::cout << "res size : " << res.size() << std::endl;
413  // std::cout << "n_slices : " << Slice->n_slices << std::endl;
414  assert((int)res.size() >= fSlices->n_slices);
415 
416  res.resize(fSlices->n_slices);
417 
418  std::transform(res.begin(), res.end(), res.begin(),
419  std::bind1st(std::multiplies<double>(), factor));
420 
421  fInducedVoltage = res;
422 
423  if (length > 0) {
424  if (length > res.size())
425  res.resize(length, 0);
426  else
427  res.resize(length);
428  }
429 
430  return res;
431  }
432 }
