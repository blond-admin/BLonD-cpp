167  static inline fft_plan_t find_plan(uint n, fft_type_t type, uint threads,
168  std::vector<fft_plan_t> &v)
169  {
170  const uint flag = FFTW_FLAGS;
171  auto it =
172  std::find_if(v.begin(), v.end(), [n, type](const fft_plan_t &s) {
173  return ((s.n == n) && (s.type == type));
174  });
175 
176  if (it == v.end()) {
177  // std::cout << "I have to create a new plan :(\n";
178  fft_plan_t plan;
179  plan.n = n;
180  plan.type = type;
181 
182  if (type == FFT) {
183  fftw_complex *in =
184  (fftw_complex *)fftw_malloc(sizeof(fftw_complex) * n);
185  fftw_complex *out =
186  (fftw_complex *)fftw_malloc(sizeof(fftw_complex) * n);
187 
188  auto p = init_fft(n, reinterpret_cast<complex_t *>(in),
189  reinterpret_cast<complex_t *>(out),
190  FFTW_FORWARD, flag, threads);
191  plan.p = p;
192  plan.in = in;
193  plan.out = out;
194  } else if (type == IFFT) {
195 
196  fftw_complex *in =
197  (fftw_complex *)fftw_malloc(sizeof(fftw_complex) * n);
198  fftw_complex *out =
199  (fftw_complex *)fftw_malloc(sizeof(fftw_complex) * n);
200  auto p = init_fft(n, reinterpret_cast<complex_t *>(in),
201  reinterpret_cast<complex_t *>(out),
202  FFTW_BACKWARD, flag, threads);
203  plan.p = p;
204  plan.in = in;
205  plan.out = out;
206 
207  } else if (type == RFFT) {
208  double *in = (double *)fftw_malloc(sizeof(double) * n);
209  fftw_complex *out = (fftw_complex *)fftw_malloc(
210  sizeof(fftw_complex) * (n / 2 + 1));
211  auto p = init_rfft(n, in, reinterpret_cast<complex_t *>(out),
212  flag, threads);
213  // std::cout << "threads : " << threads << "\n";
214  plan.p = p;
215  plan.in = in;
216  plan.out = out;
217 
218  } else if (type == IRFFT) {
219  double *out = (double *)fftw_malloc(sizeof(double) * n);
220  fftw_complex *in = (fftw_complex *)fftw_malloc(
221  sizeof(fftw_complex) * (n / 2 + 1));
222  auto p = init_irfft(n, reinterpret_cast<complex_t *>(in), out,
223  flag, threads);
224  plan.p = p;
225  plan.in = in;
226  plan.out = out;
227 
228  } else {
229  std::cerr << "[fft::find_plan]: ERROR "
230  << "Wrong fft type!\n";
231  exit(-1);
232  }
233 
234  v.push_back(plan);
235  // planV.push_back(plan);
236  return plan;
237  } else {
238  return *it;
239  }

339  // Same as python's numpy.fft.rfftfreq
340  // @ n: window length
341  // @ d (optional) : Sample spacing
342  // @return: A vector of length (n div 2) + 1 of the sample frequencies
343  static inline f_vector_t rfftfreq(const uint n, const double d = 1.0)
344  {
345  f_vector_t v(n / 2 + 1);
346  const double factor = 1.0 / (d * n);
347  #pragma omp parallel for
348  for (int i = 0; i < (int)v.size(); ++i) {
349  v[i] = i * factor;
350  }
351  return v;
352  }
