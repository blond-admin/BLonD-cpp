44 void Slices::set_cuts()
45 {
46  /*
47  *Method to set the self.cut_left and self.cut_right properties. This is
48  done as a pre-processing if the mode is set to 'const_space', for
49  'const_charge' this is calculated each turn.*
50  *The frame is defined by :math:`n\sigma_{RMS}` or manually by the user.
51  If not, a default frame consisting of taking the whole bunch +5% of the
52  maximum distance between two particles in the bunch will be taken
53  in each side of the frame.*
54  */
55 
56  if (cut_left == 0 && cut_right == 0) {
57  if (n_sigma == 0) {
58  sort_particles();
59  cut_left =
60  beam->dt.front() - 0.05 * (beam->dt.back() - beam->dt.front());
61  cut_right =
62  beam->dt.back() + 0.05 * (beam->dt.back() - beam->dt.front());
63  } else {
64  double mean_coords = mymath::mean(beam->dt.data(), beam->dt.size());
65  double sigma_coords = mymath::standard_deviation(
66  beam->dt.data(), beam->dt.size(), mean_coords);
67  cut_left = mean_coords - n_sigma * sigma_coords / 2;
68  cut_right = mean_coords + n_sigma * sigma_coords / 2;
69  }
70  } else {
71  cut_left = convert_coordinates(cut_left, cuts_unit);
72  cut_right = convert_coordinates(cut_right, cuts_unit);
73  }
74 
75  mymath::linspace(edges.data(), cut_left, cut_right, n_slices + 1);
76  for (uint i = 0; i < bin_centers.size(); ++i)
77  bin_centers[i] = (edges[i + 1] + edges[i]) / 2;
78 }

80 void Slices::sort_particles()
81 {
82  /*
83  *Sort the particles with respect to their position.*
84  */
85 
86  struct API particle {
87  double dE;
88  double dt;
89  int id;
90  bool operator<(const particle &other) const
91  {
92  return dt < other.dt;
93  }
94  };
95 
96  std::vector<particle> particles(beam->dE.size());
97  for (uint i = 0; i < particles.size(); i++)
98  particles[i] = {beam->dE[i], beam->dt[i], beam->id[i]};
99 
100  std::sort(particles.begin(), particles.end());
101 
102  for (uint i = 0; i < particles.size(); i++) {
103  beam->dE[i] = particles[i].dE;
104  beam->dt[i] = particles[i].dt;

144 void Slices::histogram(const double *__restrict input,
145  double *__restrict output,
146  const double cut_left,
147  const double cut_right,
148  const int n_slices,
149  const int n_macroparticles)
150 {
151 
152  const double inv_bin_width = n_slices / (cut_right - cut_left);
153  // histogram is faster with ints
154  typedef double hist_t;
155  hist_t *h;
156  #pragma omp parallel
157  {
158  const int threads = omp_get_num_threads();
159  const int id = omp_get_thread_num();
160  int tile = (n_macroparticles + threads - 1) / threads;
161  int start = id * tile;
162  int end = std::min(start + tile, n_macroparticles);
163  const int row = id * n_slices;
164 
165  #pragma omp single
166  h = (hist_t *)calloc(threads * n_slices, sizeof(hist_t));
167 
168  hist_t *h_row = &h[row];
169 
170  for (int i = start; i < end; ++i) {
171  const double a = input[i];
172  if (a < cut_left || a > cut_right) continue;
173  const int ffbin = (int)((a - cut_left) * inv_bin_width);
174  h_row[ffbin]++;
175  }
176  #pragma omp barrier
177 
178  tile = (n_slices + threads - 1) / threads;
179  start = id * tile;
180  end = std::min(start + tile, n_slices);
181 
182  for (int i = start; i < end; i++)
183  output[i] = 0;
184  // memset(&output[start], 0, (end-start) * sizeof(double));
185 
186  for (int i = 0; i < threads; ++i) {
187  const int r = i * n_slices;
188  for (int j = start; j < end; ++j) {
189  output[j] += h[r + j];
190  }
191  }
192  }
193 
194  if (h) free(h);
195 }

213 void Slices::smooth_histogram(const double *__restrict input,
214  double *__restrict output,
215  const double cut_left,
216  const double cut_right,
217  const int n_slices,
218  const int n_macroparticles)
219 {
220 
221  int i;
222  double a;
223  double fbin;
224  double ratioffbin;
225  double ratiofffbin;
226  double distToCenter;
227  int ffbin = 0;
228  int fffbin = 0;
229  const double inv_bin_width = n_slices / (cut_right - cut_left);
230  const double bin_width = (cut_right - cut_left) / n_slices;
231 
232  for (i = 0; i < n_slices; i++) {
233  output[i] = 0;
234  }
235 
236  for (i = 0; i < n_macroparticles; i++) {
237  a = input[i];
238  if ((a < (cut_left + bin_width * 0.5)) ||
239  (a > (cut_right - bin_width * 0.5)))
240  continue;
241  fbin = (a - cut_left) * inv_bin_width;
242  ffbin = (int)(fbin);
243  distToCenter = fbin - (double)(ffbin);
244  if (distToCenter > 0.5)
245  fffbin = (int)(fbin + 1.0);
246  ratioffbin = 1.5 - distToCenter;
247  ratiofffbin = 1 - ratioffbin;
248  if (distToCenter < 0.5)
249  fffbin = (int)(fbin - 1.0);
250  ratioffbin = 0.5 - distToCenter;
251  ratiofffbin = 1 - ratioffbin;
252  output[ffbin] = output[ffbin] + ratioffbin;
253  output[fffbin] = output[fffbin] + ratiofffbin;
254  }
255 }

266 void Slices::rms()
267 {
268  /*
269  * Computation of the RMS bunch length and position from the line density
270  (bunch length = 4sigma).*
271  */
272 
273  const auto timeResolution = bin_centers[1] - bin_centers[0];
274  const auto trap = mymath::trapezoid(n_macroparticles, timeResolution);
275 
276  auto lineDen = n_macroparticles / trap;
277  bp_rms = mymath::trapezoid(bin_centers * lineDen, timeResolution);
278 
279  auto temp = mymath::trapezoid(
280  apply_f(bin_centers - bp_rms,
281  [](double x) {return x * x; }) * lineDen, timeResolution);
282 
283  bl_rms = 4 * std::sqrt(temp);
284 }

287 void Slices::fwhm(const double shift)
288 {
289 
290  /*
291  * Computation of the bunch length and position from the FWHM
292  assuming Gaussian line density.*
293  */
294  double max = *max_element(ALL(n_macroparticles));
295  double half_max = shift + 0.5 * (max - shift);
296  double timeResolution = bin_centers[1] - bin_centers[0];
297  // First aproximation for the half maximum values
298  int taux1, taux2;
299 
300  int i = 0;
301  while (i < n_slices && n_macroparticles[i] < half_max) i++;
302  taux1 = i;
303  // prev1 is one before taux1, if taux1 is 0, then prev1 is the last
304  int prev1 = taux1 > 0 ? taux1 - 1 : n_slices - 1;
305 
306  i = n_slices - 1;
307  while (i >= 0 && n_macroparticles[i] < half_max) i--;
308  taux2 = i;
309 
310  double t1, t2;
311 
312  if (taux1 < n_slices && taux2 < n_slices - 1 && taux2 >= 0) {
313  t1 = bin_centers[taux1] -
314  (n_macroparticles[taux1] - half_max) /
315  (n_macroparticles[taux1] - n_macroparticles[prev1]) *
316  timeResolution;
317 
318  t2 = bin_centers[taux2] +
319  (n_macroparticles[taux2] - half_max) /
320  (n_macroparticles[taux2] - n_macroparticles[taux2 + 1]) *
321  timeResolution;
322  bl_fwhm = 4 * (t2 - t1) / cfwhm;
323  bp_fwhm = (t1 + t2) / 2;
324  } else {
325  bl_fwhm = nan("");
326  bp_fwhm = nan("");
327  }
328 }

373 void Slices::beam_profile_derivative(f_vector_t &x,
374  f_vector_t &derivative,
375  std::string mode)
376 {
377  /*
378  The input is one of the two available methods for differentiating
379  a function. The two outputs are the coordinate step and the discrete
380  derivative of the Beam profile respectively.
381  */
382 
383  x = bin_centers;
384  const auto dist_centers = x[1] - x[0];
385  if (mode == "filter1d") {
386  derivative = gaussian_filter1d(n_macroparticles, 1, 1, "wrap");
387  for (auto &d : derivative) d /= dist_centers;
388 
389  } else if (mode == "gradient") {
390  derivative = gradient(n_macroparticles, dist_centers);
391 
392  } else if (mode == "diff") {
393  derivative.resize(n_macroparticles.size() - 1);
394  for (uint i = 1; i < n_macroparticles.size(); i++)
395  derivative[i - 1] = (n_macroparticles[i] - n_macroparticles[i - 1])
396  / dist_centers;
397  f_vector_t diffCenters(x.begin(), x.end() - 1);
398  diffCenters += dist_centers / 2.;
399  derivative = mymath::interp(x, diffCenters, derivative);
400  } else {
401  std::cerr << "Option for derivative is not recognized.\n";
402  exit(-1);
403  }
404 
405 }
