70  inline double fast_exp(double initial_x)
71  {
72 
73  double x = initial_x;
74  double px = details::fpfloor(details::LOG2E * x + 0.5);
75 
76  const int32_t n = int32_t(px);
77 
78  x -= px * 6.93145751953125E-1;
79  x -= px * 1.42860682030941723212E-6;
80 
81  const double xx = x * x;
82 
83  // px = x * P(x**2).
84  px = details::PX1exp;
85  px *= xx;
86  px += details::PX2exp;
87  px *= xx;
88  px += details::PX3exp;
89  px *= x;
90 
91  // Evaluate Q(x**2).
92  double qx = details::QX1exp;
93  qx *= xx;
94  qx += details::QX2exp;
95  qx *= xx;
96  qx += details::QX3exp;
97  qx *= xx;
98  qx += details::QX4exp;
99 
100  // e**x = 1 + 2x P(x**2)/( Q(x**2) - P(x**2) )
101  x = px / (qx - px);
102  x = 1.0 + 2.0 * x;
103 
104  // Build 2^n in double.
105  x *= details::uint642dp((((uint64_t)n) + 1023) << 52);
106 
107  if (initial_x > details::EXP_LIMIT)
108  x = std::numeric_limits<double>::infinity();
109  if (initial_x < -details::EXP_LIMIT)
110  x = 0.;
111 
112  return x;
113  }

119  inline float fast_expf(float initial_x)
120  {
121 
122  float x = initial_x;
123 
124  float z =
125  details::fpfloor(details::LOG2EF * x +
126  0.5f); /* floor() truncates toward -infinity. */
127 
128  x -= z * details::C1F;
129  x -= z * details::C2F;
130  const int32_t n = int32_t(z);
131 
132  const float x2 = x * x;
133 
134  z = x * details::PX1expf;
135  z += details::PX2expf;
136  z *= x;
137  z += details::PX3expf;
138  z *= x;
139  z += details::PX4expf;
140  z *= x;
141  z += details::PX5expf;
142  z *= x;
143  z += details::PX6expf;
144  z *= x2;
145  z += x + 1.0f;
146 
147  /* multiply by power of 2 */
148  z *= details::uint322sp((n + 0x7f) << 23);
149 
150  if (initial_x > details::MAXLOGF)
151  z = std::numeric_limits<float>::infinity();
152  if (initial_x < details::MINLOGF)
153  z = 0.f;
154 
155  return z;
156  }
