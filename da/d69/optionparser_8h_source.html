1827  ++i2;

1834  }
1835  }

1843  // longopt in the middle
1844  // of shortopt group
1845

1847  while (*optarg != 0 && *optarg != '=')

1850  ++optarg;

1854  }

1860  break; // end of short option group

1865  ++idx;
1866

1869  else
1870  // if the potential argument is attached

1872  }
