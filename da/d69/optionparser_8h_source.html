1726

1729  // a lone minus character is a non-option argument

1745  if (numargs > 0)

1753  handle_short_options = false;
1754  longopt_name = param + 2;

1758  // -long-option

1766  // is executed only once

1774  while (usage[idx].longopt != 0 &&
1775  !streq(usage[idx].longopt, longopt_name))
1776  ++idx;
1777

1788  // for another match

1790  while (usage[i2].longopt != 0 &&
1791  !streqabbr(usage[i2].longopt, longopt_name,
1792  min_abbr_len))
1793  ++i2;

1796  // second match it's
1797  // unambiguous, so

1806  handle_short_options = false;
1807

1809  // longopt in the middle
1810  // of shortopt group
1811

1814  ++optarg;

1816  ++optarg;
1817  else

1819  optarg = (have_more_args ? args[1] : 0);
1820  }
1821

1823  * ***********************************/
1824  if (handle_short_options) {
1825  if (*++param == 0) // point at the 1st/next option character
1826  break; // end of short option group
1827 
1828  idx = 0;
1829  while (usage[idx].shortopt != 0 &&
1830  !instr(*param, usage[idx].shortopt))
1831  ++idx;
1832 
1833  if (param[1] == 0) // if the potential argument is separate
1834  optarg = (have_more_args ? args[1] : 0);
1835  else
1836  // if the potential argument is attached
1837  optarg = param + 1;
1838  }

1841

1845  // to use as Descriptor for unknown options
1846  idx = 0;

1848  (usage[idx].shortopt[0] != 0 ||
1849  usage[idx].longopt[0] != 0))
1850  ++idx;
1851  descriptor = (usage[idx].shortopt == 0 ? 0 : &usage[idx]);
1852  }
1853

1856  switch (descriptor->check_arg(option, print_errors)) {
1857  case ARG_ILLEGAL:

1861  // separated argument

2663  static void printUsage(IStringWriter& write, const Descriptor usage[],
2664  int width = 80, //
2665  int last_column_min_percent = 50,
2666  int last_column_own_line_max_percent = 75) {

2668  width = 80;
2669 
2670  if (width >
2671  10000) // protect against overflow in the following computation
2672  width = 10000;
2673

2675  ((width * last_column_min_percent) + 50) / 100;
2676  int last_column_own_line_max_width =
2677  ((width * last_column_own_line_max_percent) + 50) / 100;
2678  if (last_column_own_line_max_width == 0)
2679  last_column_own_line_max_width = 1;

2682  while (part.nextTable()) {
2683

2685  * *******************************/

2688  int col_width[maxcolumns];
2689  int lastcolumn;
2690  int leftwidth;
2691  int overlong_column_threshold = 10000;
2692  do {
2693  lastcolumn = 0;
2694  for (int i = 0; i < maxcolumns; ++i)
2695  col_width[i] = 0;
2696

2699  while (part.next()) {

2721  * line as the other columns, we can fix that by starting it
2722  * on its own line.
2723  * However we can't do this for any of the columns
2724  * 0..lastcolumn-1.
2725  * If their sum exceeds the maximum width we try to fix this
2726  * by iteratively
2727  * ignoring the widest line parts in the width determination
2728  * until
2729  * we arrive at a series of column widths that fit into one

2732  * formatted
2733  * except for a few overlong fragments.
2734  * */

2737  overlong_column_threshold = 0;

2759  // overlong_column_threshold
