17 void PhaseNoise::spectrum_to_phase_noise(f_vector_t &t, f_vector_t &dphi,
18  const f_vector_t &freq_array,
19  const f_vector_t &ReS,
20  PhaseNoise::transform_t transform)
21 {
22 
23  // Resolution in time domain
24  const auto ReSLen = ReS.size();
25  const auto f_max = freq_array.back();
26  uint nt = 0;
27  uint dt = 0;
28 
29  if (transform == transform_t::transform_none ||
30  transform == transform_t::r) {
31  nt = 2 * (ReSLen - 1);
32  dt = 1 / (2 * f_max);
33  } else if (transform == transform_t::c) {
34  nt = ReSLen;
35  dt = 1 / f_max;
36  } else {
37  std::cerr << "ERROR: The choise of Fourier transform for the\n"
38  << "RF noise generation could not be recognized.\n"
39  << "Use 'r' or 'c'.\n";
40  exit(-1);
41  }
42 
43  // Generate white noise in time domain
44  f_vector_t r1(nt);
45  f_vector_t r2(nt);
46 
47  if (fSeed1 < 0 || fSeed2 < 0) {
48  f_vector_t random;
49  // std::string home = util::GETENV("TEST_FILES");
50  util::read_vector_from_file(random, TEST_FILES "/normal_distribution.dat");
51  for (uint i = 0; i < r1.size(); i++) {
52  r1[i] = random[(2 * i) % random.size()];
53  r2[i] = random[(2 * i + 1) % random.size()];
54  }
55  } else {
56  std::default_random_engine gen(fSeed1);
57  std::uniform_real_distribution<> dist(0, 1);
58  for (auto &v : r1) v = dist(gen);
59  gen.seed(fSeed2);
60  for (auto &v : r2) v = dist(gen);
61  }
62 
63  if (transform == transform_t::transform_none ||
64  transform == transform_t::r) {
65  f_vector_t Gt(nt);
66 
67  auto factor = 2 * constant::pi;
68  auto f1 = [factor](double x) { return std::cos(factor * x); };
69  std::transform(r1.begin(), r1.end(), r1.begin(), f1);
70 
71  auto f2 = [](double x) { return std::sqrt(-2 * std::log(x)); };
72  std::transform(r2.begin(), r2.end(), r2.begin(), f2);
73 
74  std::transform(r1.begin(), r1.end(), r2.begin(), Gt.begin(),
75  std::multiplies<double>());
76 
77  // FFT to frequency domain
78  complex_vector_t Gf;
79  fft::rfft(Gt, Gf);
80 
81  // Multiply by desired noise probability density
82  factor = 2 * f_max;
83  auto f3 = [factor](double x) { return std::sqrt(factor * x); };
84 
85  r1.resize(ReS.size());
86  std::transform(ReS.begin(), ReS.end(), r1.begin(), f3);
87 
88  auto f4 = [](double r, complex_t z) { return r * z; };
89  std::transform(r1.begin(), r1.end(), Gf.begin(), Gf.begin(), f4);
90 
91  // fft back to time domain to get final phase shift
92  Gt.clear();
93  fft::irfft(Gf, Gt);
94 
95  // Use only real part of the phase shift and normalize
96  t.resize(nt);
97  mymath::linspace(t.data(), 0, nt * dt, nt);
98 
99  auto f6 = [](complex_t z) { return z.real(); };
100 
101  dphi.resize(Gt.size());
102 
103  std::transform(Gt.begin(), Gt.end(), dphi.begin(), f6);
104 
105  } else if (transform == transform_t::c) {
106 
107  complex_vector_t Gt(nt);
108 
109  const complex_t factor = 2 * constant::pi * complex_t(0, 1);
110  auto f1 = [factor](double x) { return std::exp(factor * x); };
111  std::transform(r1.begin(), r1.end(), Gt.begin(), f1);
112 
113  auto f2 = [](double x) { return std::sqrt(-2 * std::log(x)); };
114  std::transform(r2.begin(), r2.end(), r2.begin(), f2);
115 
116  auto f3 = [](complex_t a, double b) { return a * b; };
117  std::transform(Gt.begin(), Gt.end(), r2.begin(), Gt.begin(), f3);
118 
119  // FFT to frequency domain
120  complex_vector_t Gf;
121  fft::fft(Gt, Gf);
122 
123  // Multiply by desired noise probability density
124 
125  auto factor2 = f_max;
126  auto f4 = [factor2](double x) { return std::sqrt(factor2 * x); };
127 
128  r1.resize(ReS.size());
129  std::transform(ReS.begin(), ReS.end(), r1.begin(), f4);
130 
131  auto f5 = [](double r, complex_t z) { return r * z; };
132 
133  std::transform(r1.begin(), r1.end(), Gf.begin(), Gf.begin(), f5);
134  // fft back to time domain to get final phase shift
135  Gt.clear();
136  fft::ifft(Gf, Gt);
137 
138  t.resize(nt);
139  mymath::linspace(t.data(), 0, nt * dt, nt);
140 
141  auto f6 = [](complex_t z) { return z.real(); };
142 
143  dphi.resize(Gt.size());
144 
145  std::transform(Gt.begin(), Gt.end(), dphi.begin(), f6);
146  }

149 f_vector_t PhaseNoise::spectrum_generation(int k, int nt, double df, double ampl,
150  f_vector_t freq,
151  predistortion_t predistortion)
152 {
153  int nmin = std::floor(fFMin * fFs[k] / df);
154  int nmax = std::ceil(fFMax * fFs[k] / df);
155  int nf = nt / 2 + 1;
156 
157  f_vector_t spectrum;
158 
159  if (fPredistortion == predistortion_t::exponential) {
160  spectrum.resize(nmin, 0);
161 
162  auto v1 = mymath::arange<double>(0, nmax - nmin + 1);
163 
164  auto f1 = [ampl, nmax, nmin](double x) {
165  return ampl * std::exp(std::log(100.0) * x / (nmax - nmin));
166  };
167 
168  std::transform(v1.begin(), v1.end(), v1.begin(), f1);
169 
170  spectrum.insert(spectrum.end(), v1.begin(), v1.end());
171 
172  spectrum.resize(nf, 0);
173  } else if (fPredistortion == predistortion_t::linear) {
174  spectrum.resize(nmin, 0);
175 
176  f_vector_t v1(nmax - nmin + 1);
177  mymath::linspace(v1.data(), 0, ampl, nmax - nmin + 1);
178 
179  spectrum.insert(spectrum.end(), v1.begin(), v1.end());
180 
181  spectrum.resize(nf, 0);
182  } else if (fPredistortion == predistortion_t::hyperbolic) {
183  spectrum.resize(nmin, 0);
184 
185  auto v1 = mymath::arange<double>(nmin, nmax + 1);
186 
187  auto f1 = [ampl, nmax, nmin](double x) {
188  return ampl * 1.0 / (1 + 0.99 * (nmin - x) / (nmax - nmin));
189  };
190 
191  std::transform(v1.begin(), v1.end(), v1.begin(), f1);
192  spectrum.insert(spectrum.end(), v1.begin(), v1.end());
193 
194  spectrum.resize(nf, 0);
195  } else if (fPredistortion == predistortion_t::weightfunction) {
196  spectrum.resize(nmin, 0);
197 
198  // frequency relative to fs0
199  f_vector_t frel(&freq[nmin], &freq[nmax + 1]);
200 
201  std::transform(frel.begin(), frel.end(), frel.begin(),
202  std::bind2nd(std::divides<double>(), fFs[k]));
203 
204  // truncate center freqs
205  auto f1 = [](double x) { return x > 0.999 ? 0.999 : x; };
206  std::transform(frel.begin(), frel.end(), frel.begin(), f1);
207  // rms bunch length in rad corresponding to 1.2 ns
208  auto sigma = 0.754;
209  auto gamma = 0.577216;
210  auto pi = constant::pi;
211  auto f2 = [sigma, gamma, pi](double x) {
212  auto sigma2 = sigma * sigma;
213  return std::pow(4 * pi * x / sigma2, 2) *
214  std::exp(-16 * (1 - x) / sigma2) +
215  0.25 * pow(1 +
216  8 * x / sigma2 *
217  std::exp(-8 * (1 - x) / sigma2) *
218  (gamma + std::log(8 * (1 - x) / sigma2) +
219  8 * (1 - x) / sigma2),
220  2);
221  };
222 
223  const auto factor = ampl / f2(frel[0]);
224  for (uint i = 0; i < frel.size(); ++i) {
225  frel[i] = factor * f2(frel[i]);
226  // frel[i] *= factor;
227  }
228 
229  spectrum.insert(spectrum.end(), frel.begin(), frel.end());
230 
231  spectrum.resize(nf, 0);
232 
233  } else {
234  spectrum.resize(nmin, 0);
235  spectrum.resize(nmax + 1, ampl);
236  spectrum.resize(nf, 0);
237  }
238 
239  return spectrum;

275 void PSBPhaseNoiseInjection::generate()
276 {
277  auto GP = Context::GP;
278 
279  for (uint i = 0; i < fNTurns / fCorr; ++i) {
280 
281  // Scale amplitude to keep area (phase noise amplitude) constant
282  uint k = i * fCorr; // Current time step
283  double f_max = GP->f_rev[k] / 2;
284  double f_s0 = fFs[k];
285 
286  int n_points_pos_f_incl_zero = (int)(f_max / fDeltaF) + 2;
287  int nt = 2 * (n_points_pos_f_incl_zero - 1);
288  nt = mymath::next_regular(nt);
289  n_points_pos_f_incl_zero = nt / 2 + 1;
290  double df = f_max / (n_points_pos_f_incl_zero - 1);
291 
292  double ampl = 0.0;
293  if (fRescaleAmpl == rescale_ampl_t::with_sync_freq)
294  ampl = fAi * fFs[0] / f_s0;
295  else if (fRescaleAmpl == rescale_ampl_t::no_scaling)
296  ampl = fAi;
297  else {
298  std::cerr << "ERROR: Illegal Value\n"
299  << "fRescaleAmpl should be one of "
300  << "with_sync_freq or no_scaling\n";
301  }
302 
303  f_vector_t freq(n_points_pos_f_incl_zero);
304  mymath::linspace(freq.data(), 0, f_max, n_points_pos_f_incl_zero);
305 
306  auto spectrum =
307  spectrum_generation(k, nt, df, ampl, freq, fPredistortion);
308 
309  f_vector_t noise_t, noise_dphi;
310  spectrum_to_phase_noise(noise_t, noise_dphi, freq, spectrum);
311 
312  fSeed1 += 239;
313  fSeed2 += 158;
314 
315  // Fill phase noise array
316 
317  const uint kmax =
318  i < fNTurns / fCorr - 1 ? (i + 1) * fCorr : fNTurns + 1;
319 
320  std::copy(noise_dphi.begin(), noise_dphi.begin() + kmax - k,
321  fDphi.begin() + k);
322 
323  auto rms_noise =
324  mymath::standard_deviation(noise_dphi.data(), noise_dphi.size());
325  // std::cout << "RF noise for time step " << noise_t[1]
326  // << " s (iter " << i << ") has r.m.s phase "
327  // << rms_noise << " rad (" << rms_noise * 180 / constant::pi
328  // << " deg)\n";
329  }
