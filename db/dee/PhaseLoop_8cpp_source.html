15 void PhaseLoop::beam_phase()
16 {
17  /*
18  *Beam phase measured at the main RF frequency and phase. The beam is
19  convolved with the window function of the band-pass filter of the
20  machine. The coefficients of sine and cosine components determine the
21  beam phase, projected to the range -Pi/2 to 3/2 Pi. Note that this beam
22  phase is already w.r.t. the instantaneous RF phase.*
23  */
24 
25  // Main RF frequency at the present turn
26  // omega_RF = self.rf_params.omega_RF[0,self.rf_params.counter[0]]
27  auto RfP = Context::RfP;
28  auto Slice = Context::Slice;
29 
30  double omega_rf = RfP->omega_rf[RfP->section_index][RfP->counter];
31  double phi_rf = RfP->phi_rf[RfP->section_index][RfP->counter];
32  // Convolve with window function
33  //
34  double *base = new double[Slice->n_slices];
35  double *array = new double[Slice->n_slices];
36 
37  #pragma omp parallel for
38  for (int i = 0; i < (int)Slice->n_slices; ++i) {
39  const double a = alpha * Slice->bin_centers[i];
40  base[i] = std::exp(a) * Slice->n_macroparticles[i];
41  }
42 
43  #pragma omp parallel for
44  for (int i = 0; i < (int)Slice->n_slices; ++i) {
45  const double a = omega_rf * Slice->bin_centers[i] + phi_rf;
46  array[i] = base[i] * std::sin(a);
47  }
48  double scoeff =
49  mymath::trapezoid(array, Slice->bin_centers.data(), Slice->n_slices);
50 
51  #pragma omp parallel for
52  for (int i = 0; i < (int)Slice->n_slices; ++i) {
53  const double a = omega_rf * Slice->bin_centers[i] + phi_rf;
54  array[i] = base[i] * std::cos(a);
55  }
56 
57  double ccoeff =
58  mymath::trapezoid(array, Slice->bin_centers.data(), Slice->n_slices);
59 
60  phi_beam = std::atan(scoeff / ccoeff) + constant::pi;
61 
62  delete[] base;
63  delete[] array;
64 }

86 void PhaseLoop::radial_steering_from_freq()
87 {
88  // Frequency and phase change for the current turn due to the
89  // radial steering program.
90  auto RfP = Context::RfP;
91  auto GP = Context::GP;
92  const int counter = RfP->counter;
93 
94  const auto radial_steering_domega_rf =
95  -RfP->omega_rf_d[0][counter] * RfP->eta_0[counter] / GP->alpha[0][0] *
96  reference / GP->ring_radius;
97 
98  for (int i = 0; i < RfP->n_rf; ++i) {
99  RfP->omega_rf[i][counter] += radial_steering_domega_rf *
100  RfP->harmonic[i][counter] /
101  RfP->harmonic[0][counter];
102  }
103 
104  // Update the RF phase of all systems for the next turn
105  // Accumulated phase offset due to PL in each RF system
106  for (int i = 0; i < RfP->n_rf; ++i) {
107  RfP->dphi_rf_steering[i] +=
108  2 * constant::pi * RfP->harmonic[i][counter] *
109  (RfP->omega_rf[i][counter] - RfP->omega_rf_d[i][counter]) /
110  RfP->omega_rf_d[i][counter];
111  }
112 
113  // Total phase offset
114  for (int i = 0; i < RfP->n_rf; ++i) {
115  RfP->phi_rf[i][counter] += RfP->dphi_rf_steering[i];

211 void LHC::track()
212 {
213  /*
214  Calculation of the LHC RF frequency correction from the phase difference
215  between beam and RF (actual synchronous phase). The transfer function is
216 
217  .. math::
218  \\Delta \\omega_{rf}^{PL} = - g_{PL} (\\Delta\\varphi_{PL} + \\phi_{N})
219 
220  where the phase noise for the controlled blow-up can be optionally
221  activated.
222  Using 'gain2', a synchro loop can be activated in addition to remove
223  long-term frequency drifts:
224 
225  .. math::
226  \\Delta \\omega_{rf}^{SL} = - g_{SL} (y + a \\Delta\\varphi_{rf}) ,
227 
228  where we use the recursion
229 
230  .. math::
231  y_{n+1} = (1 - \\tau) y_n + (1 - a) \\tau \\Delta\\varphi_{rf} ,
232 
233  with a and \tau being defined through the synchrotron frequency f_s and
234  the synchrotron tune Q_s as
235 
236  .. math::
237  a (f_s) \\equiv 5.25 - \\frac{f_s}{\\pi 40~\\text{Hz}} ,
238 
239  .. math::
240  \\tau(f_s) \\equiv 2 \\pi Q_s \\sqrt{ \\frac{a}{1 + \\frac{g_{PL}}{g_{SL}}
241  \\sqrt{\\frac{1 + 1/a}{1 + a}} }}
242  */
243  auto RfP = Context::RfP;
244 
245  uint counter = RfP->counter;
246  double dphi_rf = RfP->dphi_rf[0];
247 
248  beam_phase();
249  phase_difference();
250 
251  // Frequency correction from phase loop and synchro loop
252 
253  domega_rf = -gain[counter] * dphi -
254  gain2 * (lhc_y + lhc_a[counter] * (dphi_rf + reference));
255 
256  // Update recursion variable
257  lhc_y = (1 - lhc_t[counter]) * lhc_y +
258  (1 - lhc_a[counter]) * lhc_t[counter] * (dphi_rf + reference);
259 
260  default_track();
261 }

322 void PSB::track()
323 {
324  /*
325  Calculation of the PSB RF frequency correction from the phase difference
326  between beam and RF (actual synchronous phase). The transfer function is
327 
328  .. math::
329  \\Delta \\omega_{RF} = g(t) \\frac{a_0 \\Delta\\Phi_{PL}^2 + a_1
330  \\Delta\\Phi_{PL} + a_2 }{b_0 \\Delta\\Phi_{PL}^2 + b_1 \\Delta\\Phi_{PL} +
331  b_2}
332 
333  Input g through gain and [a_0, a_1, a_2, b_0, b_1, b_2] through
334  coefficients.
335  */
336  auto GP = Context::GP;
337  auto RfP = Context::RfP;
338 
339  // Average phase error while frequency is updated
340  uint counter = RfP->counter;
341 
342  beam_phase();
343  phase_difference();
344  dphi_av += dphi;
345  t_accum += GP->t_rev[counter];
346 
347  // dprintf("Before if counter is %d\n", counter);
348  // Phase loop active on certain turns
349 
350  if (counter == on_time[PL_counter] && counter > delay) {
351  dphi_av /= (on_time[PL_counter] - on_time[PL_counter - 1]);
352 
353  domega_PL = 0.998 * domega_PL -
354  gain[counter] * (dphi_av - dphi_av_prev + reference);
355 
356  // Update averaging variables
357  dphi_av_prev = dphi_av;
358  dphi_av = 0;
359 
360  // Add correction from radial loop
361  if (PL_counter % static_cast<int>(dt[1]) == 0) {
362 
363  drho = (RfP->omega_rf[0][counter] - RfP->omega_rf_d[0][counter]) /
364  (RfP->omega_rf_d[0][counter] *
365  (1 / (GP->alpha[0][0] * RfP->gamma[counter] *
366  RfP->gamma[counter]) -
367  1)) +
368  reference;
369 
370  domega_RL = domega_RL - gain2[0] * (drho - drho_prev) -
371  gain2[1] * drho * t_accum;
372 
373  drho_prev = drho;
374  t_accum = 0;
375  }
376 
377  // Counter to pick the next time step when the PL will be active
378  PL_counter++;
379  }
380  // Apply frequency correction
381  domega_rf = domega_PL + domega_RL;
382 
383  default_track();
384 }

386 void PSB::precalculate_time()
387 {
388  /*
389  For machines like the PSB, where the PL acts only in certain time
390  intervals, pre-calculate on which turns to act.
391  */
392  auto GP = Context::GP;
393 
394  uint n = delay + 1;
395 
396  while (n < GP->t_rev.size()) {
397  auto summa = 0.0;
398  while (summa < dt[0]) {
399  if (n < GP->t_rev.size()) {
400  summa += GP->t_rev[n];
401  n++;
402  } else {
403  on_time.push_back(0);
404  return;
405  }
406  }
407  on_time.push_back(n - 1);
408  }
409 }
