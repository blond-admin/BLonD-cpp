35 void Beams::statistics()
36 {
37  ftype m_dE, m_dt, s_dE, s_dt;
38  m_dt = m_dE = s_dE = s_dt = 0;
39  uint n = 0;
40  for (int i = 0; i < (int) n_macroparticles; ++i) {
41  if (id[i] != 0) {
42  m_dE += dE[i];
43  m_dt += dt[i];
44  n++;
45  }
46  }
47  mean_dE = m_dE /= n;
48  mean_dt = m_dt /= n;
49  for (int i = 0; i < (int) n_macroparticles; ++i) {
50  if (id[i] != 0) {
51  s_dE += (dE[i] - m_dE) * (dE[i] - m_dE);
52  s_dt += (dt[i] - m_dt) * (dt[i] - m_dt);
53  }
54  }
55  sigma_dE = std::sqrt(s_dE / n);
56  sigma_dt = std::sqrt(s_dt / n);
57 
58  epsn_rms_l = constant::pi * sigma_dE * sigma_dt; // in eVs
59 
60  // Losses
61  n_macroparticles_lost = n_macroparticles - n;
62 }
