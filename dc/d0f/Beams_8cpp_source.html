48 void Beams::statistics(const double *__restrict dE,
49  const double *__restrict dt,
50  const int *__restrict id,
51  const int size)
52 {
53  double m_dE, m_dt, s_dE, s_dt;
54  m_dt = m_dE = s_dE = s_dt = 0.0;
55  int n = 0;
56 
57  #pragma omp parallel for reduction(+:m_dE, m_dt, n)
58  for (int i = 0; i < size; ++i) {
59  m_dE += id[i] * dE[i];
60  m_dt += id[i] * dt[i];
61  n += id[i];
62  }
63 
64  mean_dE = m_dE /= n;
65  mean_dt = m_dt /= n;
66 
67  #pragma omp parallel for reduction(+:s_dE, s_dt)
68  for (int i = 0; i < size; ++i) {
69  s_dE += id[i] * (dE[i] - mean_dE) * (dE[i] - mean_dE);
70  s_dt += id[i] * (dt[i] - mean_dt) * (dt[i] - mean_dt);
71  }
72  sigma_dE = std::sqrt(s_dE / n);
73  sigma_dt = std::sqrt(s_dt / n);
74  epsn_rms_l = constant::pi * sigma_dE * sigma_dt; // in eVs
75  // Losses
76  n_macroparticles_lost = n_macroparticles - n;
77 
78 }
