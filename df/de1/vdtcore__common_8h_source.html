165  inline double getMantExponent(const double x, double &fe)
166  {
167 
168  uint64_t n = dp2uint64(x);
169 
170  // Shift to the right up to the beginning of the exponent.
171  // Then with a mask, cut off the sign bit
172  uint64_t le = (n >> 52);
173 
174  // chop the head of the number: an int contains more than 11 bits
175  // (32)
176  int32_t e =
177  le; // This is important since sums on uint64_t do not vectorise
178  fe = e - 1023;
179 
180  // This puts to 11 zeroes the exponent
181  n &= 0x800FFFFFFFFFFFFFULL;
182  // build a mask which is 0.5, i.e. an exponent equal to 1022
183  // which means *2, see the above +1.
184  const uint64_t p05 = 0x3FE0000000000000ULL; // dp2uint64(0.5);
185  n |= p05;
186 
187  return uint642dp(n);
188  }
