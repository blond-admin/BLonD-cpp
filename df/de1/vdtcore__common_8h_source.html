153  inline double getMantExponent(const double x, double& fe) {
154 
155  uint64_t n = dp2uint64(x);
156 
157  // Shift to the right up to the beginning of the exponent.
158  // Then with a mask, cut off the sign bit
159  uint64_t le = (n >> 52);
160 
161  // chop the head of the number: an int contains more than 11 bits
162  // (32)
163  int32_t e =
164  le; // This is important since sums on uint64_t do not vectorise
165  fe = e - 1023;
166 
167  // This puts to 11 zeroes the exponent
168  n &= 0x800FFFFFFFFFFFFFULL;
169  // build a mask which is 0.5, i.e. an exponent equal to 1022
170  // which means *2, see the above +1.
171  const uint64_t p05 = 0x3FE0000000000000ULL; // dp2uint64(0.5);
172  n |= p05;
173 
174  return uint642dp(n);
175  }
